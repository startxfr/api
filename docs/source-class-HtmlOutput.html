 99:     /**
100:      * Recursively render an array to an HTML list and details for folding results
101:      *
102:      * @param   mixed   data to be rendered and returned to the client
103:      * @return string
104:      */
105:     protected function layoutContent($content) {
106:         if (is_string($content)) {
107:             return $content;
108:         }
109:         $ech = "<ul>\n";
110:         if (!is_array($content))
111:             $content = array($content);
112:         elseif (is_object($content))
113:             $content = (array) $content;
114:         foreach ($content as $field => $value) {
115:             if (is_object($value) and $value instanceof MongoDate)
116:                 $value = date('Y-m-d H:i:s', $value->sec);
117:             elseif (is_object($value))
118:                 $value = (array) $value;
119:             if (is_array($value)) {
120:                 $ech .= "<details open><summary>" . $field . "</summary> ";
121:                 $ech .= $this->layoutContent($value, '');
122:                 $ech .= "</details>";
123:             } else {
124:                 $ech .= "<li><strong>" . $field . "</strong> ";
125:                 $value = htmlentities($value, ENT_COMPAT, 'UTF-8');
126:                 if ((strpos($value, 'http://') === 0) || (strpos($value, 'https://') === 0))
127:                     $ech .= "<a href=\"" . $value . "\">" . $value . "</a>";
128:                 else
129:                     $ech .= $value;
130:                 $ech .= "</li>\n";
131:             }
132:         }
133:         $ech .= "</ul>\n";
134:         return $ech;
135:     }
